    % Place nodes
    \node [block] (S) {$S$};
    \node [block] at (3,0) (E) {$E$};

    \node [block] at (9,0) (R) {$R$};

    \node (dE) [below of=E,node distance=2cm, coordinate] {E};
    \node (dI) [below of=I,node distance=2cm, coordinate] {I};
    \node (dR) [below of=R,node distance=2cm, coordinate] {R};
    \node (bS) [left of=S,node distance=2cm, coordinate] {S};
    % Draw edges
    \path [line] (S) to node {$\beta I/N$} (E);
    \path [line] (E) to node {$\delta$} (I);
    \path [line] (I) to node {$\gamma$} (R);
    \path [line] (S) to node {$\mu$} (dS);
    \path [line] (E) to node {$\mu$} (dE);
    \path [line] (I) to node {$\mu$} (dI);
